\begin{tabular}{@{}l @{\ \ } l@{}}

\bottomrule

% ---- RIGHT: the kernel ---------------------------------------------------
\begin{column}{0.58\textwidth}
\begin{block}{Cycle time}
\[
  \tcyc = \max\!\Big(
    \tdec,\; \tsa+\tsw,\;
    \tfrac{\tdec+\twl+\tbl+\tsa+\tsw}{\Nshare}
  \Big)
\]
\end{block}

\vspace{-1.0ex}
\begin{block}{Bandwidth}
\[
  \boxed{\;\mathrm{BW} = \dfrac{\bacc}{\tcyc}\,\Nindep\;}
  \qquad \bacc \le \Nbl
\]
\end{block}

\vspace{-1.0ex}
\begin{block}{Counts, volume, scaling}
\vspace{-1.0ex}
\begin{align*}
  \Ntot &= \Nindep\Nshare, \quad
  \Nsa = \Nindep\,\bacc, \quad
  \Ndec = \Nindep \\[0.3ex]
  \mathrm{Vol} &= \Volarr\Ntot + \Volsa\Nsa
                 + \Voldec\Ndec + \Volsel(\Nshare) \\[0.3ex]
  \Voldec &\propto \Nwl\Nbl \propto \Volarr, \quad
  \Volsa \propto \tfrac{1}{\mathrm{margin}^2} \\[0.3ex]
  \twl &= k^{\mathrm{w}}_{WL}\Nbl^2 + k^{\mathrm{c}}_{WL}\Nbl \quad\text{\scriptsize(wire self-RC + cell load)} \\[0.2ex]
  \tbl &= k^{\mathrm{w}}_{BL}\Nwl^2 + k^{\mathrm{c}}_{BL}\Nwl \quad\text{\scriptsize($k^{\mathrm{w}}$ shared, $k^{\mathrm{c}}$ per-tech)}
\end{align*}
\end{block}

% ==== Frame 2: cycle time, bandwidth, packing =============================
\begin{frame}{Cycle time, bandwidth, and packing}
\footnotesize

\begin{block}{Cycle time (throughput bound)}
\[
  \tcyc = \max\!\Big(
    \tdec,\; \tsa+\tsw,\;
    \tfrac{\tdec+\twl+\tbl+\tsa+\tsw}{\Nshare}
  \Big)
\]
\end{block}

\begin{block}{Bandwidth}
\[
  \boxed{\;\mathrm{BW} = \dfrac{\bacc}{\tcyc}\,\Nindep\;}
  \qquad \bacc \le \Nbl \quad\text{\scriptsize(whole-row-activate limit)}
\]
\end{block}

\begin{block}{Counts \& volume (packing equality)}
\vspace{-1.0ex}
\begin{align*}
  \Ntot &= \Nindep\Nshare, \qquad
  \Nsa = \Nindep\,\bacc, \qquad
  \Ndec = \Nindep \\[0.4ex]
  \mathrm{Vol} &= \Volarr\Ntot + \Volsa\Nsa + \Voldec\Ndec
                 + \Volsel(\Nshare) + \Volper\Ntot
                 \;\le\; A\,L\,t_{\mathrm{layer}}
\end{align*}
\end{block}

\end{frame}

% ==== Frame 3: scaling laws, sensing, energy ==============================
\begin{frame}{Scaling laws, sensing, and power density}
\footnotesize

\begin{block}{Timing scaling ($\twl$ wire/cell; $\tbl$ sense-signal settling, Upton 2024)}
\vspace{-1.0ex}
\begin{align*}
  \twl &= k^{\mathrm{w}}_{WL}\Nbl^2 + k^{\mathrm{c}}_{WL}\Nbl
        \qquad\text{\scriptsize(WL wire self-RC + cell/gate load)} \\[0.2ex]
  \tbl &= f_\Delta\big(a_{\mathrm{lin}}\Nwl + a_{\mathrm{quad}}\Nwl^2\big),
        \quad f_\Delta = -\ln(1-\Delta),
        \quad a_{\mathrm{quad}} = \tfrac12 r_{BL}c_{BL}
\end{align*}
\vspace{-1.8ex}
\begin{center}\scriptsize
\begin{tabular}{@{}l l l@{}}
\toprule
mode & $a_{\mathrm{lin}}$ (signal develop) & note \\
\midrule
current (App.\ A) & $c_{BL}\,nU_t/I_{\mathrm{read}}$ & \\
voltage (App.\ B) & $v_{\mathrm{ratio}}R_{\text{pull-up}}c_{BL}$ & $v_{\mathrm{ratio}}$ also scales $a_{\mathrm{quad}}$ \\
charge-share (DRAM) & $r_{BL}C_{\mathrm{cell}}$ & row cap $\Nwl\!\le\!\tfrac{C_{\mathrm{cell}}(1-m_{\mathrm{SA}})}{c_{BL}m_{\mathrm{SA}}}$ \\
\bottomrule
\end{tabular}
\end{center}
\end{block}

\vspace{-1.6ex}
\begin{block}{Energy \& power density (per-access $CV^2$, DESTINY)}
\vspace{-1.0ex}
\begin{align*}
  \Eacc &= k_{\mathrm{col}}\Nbl + k_{\mathrm{arr}}\Nbl\Nwl + e_{\mathrm{periph}} + f_w e_{\mathrm{write}}\bacc,
        \quad k_{\mathrm{col}}=C_{\mathrm{cell}}V^2,\; k_{\mathrm{arr}}=c_{BL}V^2 \\[0.3ex]
  \Ebit &= \Eacc/\bacc, \qquad \Pdyn = \mathrm{BW}\cdot\Ebit, \qquad
        \Aused = \mathrm{Vol}/(L\,t_{\mathrm{layer}}) \\[0.3ex]
  &\boxed{\;\big(\Pdyn + p_{\mathrm{leak}}\Ncell\big)\big/\Aused \le \Pmax\;}
\end{align*}
\end{block}

\end{frame}
